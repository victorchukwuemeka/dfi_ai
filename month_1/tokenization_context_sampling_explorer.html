# Rough token estimator (rule of thumb: 1 token ≈ 4 chars / 0.75 words) def estimate_tokens(text: str) -> dict: char_estimate = len(text) / 4 word_estimate = len(text.split()) / 0.75 return { "chars": len(text), "token_estimate": int((char_estimate + word_estimate) / 2), "cost_gpt4o_input": f"${int((char_estimate+word_estimate)/2) * 0.0000025:.6f}", } # Real-world context budgeting CONTEXT_LIMIT = 128_000 # GPT-4o SYSTEM_PROMPT_TOKENS = 300 RESPONSE_RESERVE = 2_000 # reserve for model output available_for_history = ( CONTEXT_LIMIT - SYSTEM_PROMPT_TOKENS - RESPONSE_RESERVE ) # → 125,700 tokens for conversation print(f"Max chat history tokens: {available_for_history:,}")

Sampling mechanics — how the model picks the next token

The code behind sampling
import numpy as np def sample_next_token(logits, temperature=1.0, top_p=0.9, top_k=50): """Core sampling logic used by every LLM inference engine.""" # 1. Apply temperature — sharpens or flattens the distribution logits = logits / temperature # temp→0: argmax; temp→∞: uniform # 2. Convert to probabilities probs = np.exp(logits - logits.max()) # numerically stable softmax probs = probs / probs.sum() # 3. Top-k filtering — zero out everything outside top k if top_k > 0: top_k_indices = np.argsort(probs)[-top_k:] mask = np.zeros_like(probs, dtype=bool) mask[top_k_indices] = True probs[~mask] = 0 probs /= probs.sum() # 4. Top-p (nucleus) filtering — keep tokens covering p of mass sorted_probs = np.sort(probs)[::-1] cumulative = np.cumsum(sorted_probs) cutoff = sorted_probs[np.searchsorted(cumulative, top_p)] probs[probs < cutoff] = 0 probs /= probs.sum() # 5. Sample! return np.random.choice(len(probs), p=probs) # Temperature cheat sheet: # 0.0–0.3 → deterministic, factual, code # 0.5–0.8 → balanced, conversational # 1.0–1.5 → creative, diverse # >1.5 → chaotic, often incoherent
